 CREATE TABLE SCHEDULE (
  ID INT NOT NULL AUTO_INCREMENT,
  User_ID INT NOT NULL,
  Date DATE NOT NULL,
  Room_ID INT NOT NULL,
  Start INT NOT NULL,
  End INT NOT NULL,
   PRIMARY KEY ( ID )
  )ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE USER (
  	ID INT NOT NULL AUTO_INCREMENT,
  	Card VARCHAR(20) NOT NULL,
  	Mail VARCHAR(20) NOT NULL,
  	Name VARCHAR(20) NOT NULL,
   	PRIMARY KEY ( ID )
   	)ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE ROOM (
  	ID INT NOT NULL AUTO_INCREMENT,
  	Name VARCHAR(20) NOT NULL,
   	PRIMARY KEY ( ID )
   	)ENGINE=InnoDB DEFAULT CHARSET=utf8;
